package tyust_lab;
import java.util.Scanner;
import java.util.Array;

public class Lab2 {
	private int wager = 0;

	public int get_wager() {
		return this.wager;
	}

	public void set_wager(int wager) {
		this.wager = wager;
	}
	public void outPut(int[][] array) {
		System.out.println("您下注的注数为："+ this.wager );
		int[] sortedArray;
		sortedArray = new int[7];
		for(int m=0;m < this.wager; m++) {
			sortedArray = array[m];
			Arrays.sort(sortedArray);
			for(int n=0; n< 7; n++) {
				System.out.print(sortedArray[n]+ " ");
			}
			System.out.println();
		}
	}
	public void bet(int[][] array) {
		// 产生1-35的随机整数
		int random_num = (int)(Math.random()*34+1);
		for(int m=0; m< this.wager; m++) {
			for (int i=0; i < 7; i++) {
				//如果 i==0  不对随机数进行校验
				if (i !=0) {
					out : while(true) {
						// 产生1-35的随机整数
						random_num = (int)(Math.random()*34+1);
						// 这个j要取狭义上的全局
						int j = 0 ;
						for(j=i; j>0; j--) {
							if (random_num == array[m][j-1])
								// 如果相等跳出这个for循环
								break;
						}
						//直到j==0 跳出while循环，说明找到第一个元素仍然没有出现重复。上面的break即使在中途跳出了循环但是j!=0所以他依然不能跳出死循环
						if (j==0) break out;
					}
				}
				array[m][i] = random_num;
			}
		}
		outPut(array);
	}

	public void main() {
		// 用户输入下注数
		System.out.println("尊敬的顾客你好，请输入下注的注数");
		Scanner scnn = new Scanner(System.in);
		int zhushu = scnn.nextInt();
		this.set_wager(zhushu);
		scnn.close();
		int[][] array;
		array = new int[this.wager][7];
		this.bet(array);

	}
}
